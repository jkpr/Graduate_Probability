\item[6.3.7]
Show: If $F_n \vra F$ and $G_n \vra G$, then $F_n * G_n \vra F * G$.

\begin{proof}
Let $F_n, G_n, F,$ and $G$ be d.f.'s with ch.f.'s $f_n, g_n, f,$ and $g$, respectively. By the properties of convolutions, $F_n * G_n$ and $F * G$ are d.f.'s, so let their respective p.m.'s be $\mu_n$ and $\mu$. 
Let $F_n \vra F$ and $G_n \vra G$. 
Thus by theorem 6.3.1, $f_n \to f$ and $g_n \to g$ uniformly on every finite interval. 
In particular, that convergence holds on $[t - \epsilon, t + \epsilon]$ for all $t \in \R$ and for all finite $\epsilon > 0$.
This implies
\begin{equation}
f_n \to f 
\quad \text{and} \quad
g_n \to g
\label{eq1}
\end{equation}
pointwise on $\R$.
By theorems 6.1.4 and 3.3.4, it follows that $F_n * G_n$ has ch.f. $f_ng_n$ and $F * G$ has ch.f. $fg$. 
By properties of a limit and \eqref{eq1}, $f_ng_n \to fg$ pointwise in $\R$. By the properties of characteristic functions, $fg$ is continuous at $0$. Hence by theorem 6.3.2, $\mu_n \vra \mu$, and it 
follows that $F_n * G_n \vra F * G$.
\end{proof}



\end{enumerate}
