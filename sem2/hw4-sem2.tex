\[
e^{itx} = 1 + itx + \theta \frac{(tx) ^2}{2}
\quad
\text{for $|x| > \eta$}
\]
and
\[
e^{itx} = 1 + itx - \frac{(tx)^2}{2} + \theta' \frac{|tx|^3}{6}
\quad
\text{for $|x| \leq \eta$}
\]

\begin{proof}
From pg. 179 in Chung, $\theta$ and $\theta'$ denote a ``generic'' complex number of modulus $\leq 1$.
\end{proof}
